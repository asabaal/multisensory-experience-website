        // Tab switching functionality
        function showTab(tabName) {
            // Hide all tab contents
            const tabContents = document.querySelectorAll('.tab-content');
            tabContents.forEach(content => content.classList.remove('active'));
            
            // Remove active class from all tabs
            const tabs = document.querySelectorAll('.tab');
            tabs.forEach(tab => tab.classList.remove('active'));
            
            // Show selected tab content
            document.getElementById(tabName).classList.add('active');
            
            // Add active class to clicked tab
            event.target.classList.add('active');
        }

        // Generate file tree
        function generateFileTree() {
            const fileTree = document.getElementById('fileTree');
            
            fileData.forEach((file, index) => {
                const fileItem = document.createElement('div');
                fileItem.className = 'file-item';
                fileItem.setAttribute('data-merge-status', file.merge_readiness);
                fileItem.setAttribute('data-file-path', file.file_path.toLowerCase());
                
                const fileExtension = file.file_path.split('.').pop().toLowerCase();
                const statusIcon = getStatusIcon(file.merge_readiness);
                const businessImpact = getBusinessImpactLevel(file.overall_assessment.business_impact);
                
                fileItem.innerHTML = `
                    <div class="file-header" onclick="toggleFileDetails(${index})">
                        <span class="expand-icon" id="expand-icon-${index}">‚ñ∂</span>
                        <span class="merge-status ${getStatusClass(file.merge_readiness)}">${statusIcon}</span>
                        <span class="file-name">${file.file_path}</span>
                        <span class="file-type">${fileExtension}</span>
                    </div>
                    <div class="file-details" id="file-details-${index}">
                        <div class="detail-section">
                            <div class="detail-title">Purpose</div>
                            <div class="detail-content">${file.overall_assessment.purpose}</div>
                        </div>
                        <div class="detail-section">
                            <div class="detail-title">Business Impact</div>
                            <div class="detail-content">
                                ${file.overall_assessment.business_impact}
                                <span class="impact-score ${businessImpact.class}">${businessImpact.label}</span>
                            </div>
                        </div>
                        <div class="detail-section">
                            <div class="detail-title">Risk Assessment</div>
                            <div class="detail-content">${file.overall_assessment.risk_assessment}</div>
                        </div>
                        <div class="detail-section">
                            <div class="detail-title">Merge Status</div>
                            <div class="detail-content">
                                <span class="${getStatusClass(file.merge_readiness)}">${statusIcon} ${file.merge_readiness.toUpperCase()}</span>
                            </div>
                        </div>
                    </div>
                `;
                
                fileTree.appendChild(fileItem);
            });
        }

        function getStatusIcon(status) {
            switch(status) {
                case 'ready': return 'üü¢';
                case 'conditional': return 'üü°';
                case 'not-ready': return 'üî¥';
                default: return '‚ö™';
            }
        }

        function getStatusClass(status) {
            switch(status) {
                case 'ready': return 'status-ready';
                case 'conditional': return 'status-conditional';
                case 'not-ready': return 'status-not-ready';
                default: return 'status-unknown';
            }
        }

        function getBusinessImpactLevel(impact) {
            const impactLower = impact.toLowerCase();
            if (impactLower.includes('critical') || impactLower.includes('essential') || impactLower.includes('core')) {
                return { class: 'impact-high', label: 'HIGH' };
            } else if (impactLower.includes('medium') || impactLower.includes('supports') || impactLower.includes('provides')) {
                return { class: 'impact-medium', label: 'MEDIUM' };
            } else {
                return { class: 'impact-low', label: 'LOW' };
            }
        }

        function toggleFileDetails(index) {
            const details = document.getElementById(`file-details-${index}`);
            const icon = document.getElementById(`expand-icon-${index}`);
            const header = icon.closest('.file-header');
            
            if (details.classList.contains('expanded')) {
                details.classList.remove('expanded');
                icon.classList.remove('expanded');
                header.classList.remove('expanded');
            } else {
                details.classList.add('expanded');
                icon.classList.add('expanded');
                header.classList.add('expanded');
            }
        }

        // Filter functionality
        function setupFilters() {
            const filterButtons = document.querySelectorAll('.filter-btn');
            const fileItems = document.querySelectorAll('.file-item');
            
            filterButtons.forEach(button => {
                button.addEventListener('click', () => {
                    // Remove active class from all buttons
                    filterButtons.forEach(btn => btn.classList.remove('active'));
                    // Add active class to clicked button
                    button.classList.add('active');
                    
                    const filter = button.getAttribute('data-filter');
                    
                    fileItems.forEach(item => {
                        if (filter === 'all' || item.getAttribute('data-merge-status') === filter) {
                            item.style.display = 'block';
                        } else {
                            item.style.display = 'none';
                        }
                    });
                });
            });
        }

        // Search functionality
        function setupSearch() {
            const searchInput = document.getElementById('fileSearch');
            const fileItems = document.querySelectorAll('.file-item');
            
            searchInput.addEventListener('input', (e) => {
                const searchTerm = e.target.value.toLowerCase();
                
                fileItems.forEach(item => {
                    const filePath = item.getAttribute('data-file-path');
                    if (filePath.includes(searchTerm)) {
                        item.style.display = 'block';
                    } else {
                        item.style.display = 'none';
                    }
                });
            });
        }

        // Initialize the dashboard
        document.addEventListener('DOMContentLoaded', () => {
            generateFileTree();
            setupFilters();
            setupSearch();
        });
    </script>
