    <script>
        // INJECT_COMPLETE_FILE_DATA_HERE

        // Initialize variables after fileData is defined
        let filteredFiles = [];
        let currentFilter = 'all';

        function renderFiles() {
            const container = document.getElementById('fileAnalysis');
            container.innerHTML = '';

            if (filteredFiles.length === 0) {
                container.innerHTML = '<p style="text-align: center; color: #b0b0b0; padding: 20px;">No files match the current filter.</p>';
                return;
            }

            filteredFiles.forEach((file, index) => {
                const fileElement = document.createElement('div');
                fileElement.className = 'file-analysis';
                
                const statusClass = file.merge_readiness === 'ready' ? 'badge-ready' : 
                                   file.merge_readiness === 'conditional' ? 'badge-conditional' : 'badge-not-ready';
                
                const statusText = file.merge_readiness === 'ready' ? 'Ready' : 
                                  file.merge_readiness === 'conditional' ? 'Conditional' : 'Not Ready';

                fileElement.innerHTML = `
                    <div class="file-header expandable" onclick="toggleFile(${index})">
                        <div class="file-path">${file.file_path}</div>
                        <div class="file-status">
                            <div class="status-badge ${statusClass}">${statusText}</div>
                            <span class="collapse-icon" id="icon-${index}">▼</span>
                        </div>
                    </div>
                    <div class="file-details" id="details-${index}">
                        <div class="scores">
                            <div class="score-item">
                                <div class="score-value">${file.business_impact_score || 'N/A'}</div>
                                <div class="score-label">Business Impact</div>
                            </div>
                            <div class="score-item">
                                <div class="score-value">${file.technical_risk_score || 'N/A'}</div>
                                <div class="score-label">Technical Risk</div>
                            </div>
                        </div>
                        
                        ${file.overall_assessment ? `
                            <div class="detail-section">
                                <div class="detail-title">Purpose</div>
                                <div class="detail-content">${file.overall_assessment.purpose || 'Not specified'}</div>
                            </div>
                            
                            <div class="detail-section">
                                <div class="detail-title">Business Impact</div>
                                <div class="detail-content">${file.overall_assessment.business_impact || 'Not specified'}</div>
                            </div>
                            
                            <div class="detail-section">
                                <div class="detail-title">Risk Assessment</div>
                                <div class="detail-content">${file.overall_assessment.risk_assessment || 'Not specified'}</div>
                            </div>
                        ` : ''}
                        
                        ${file.detailed_feedback ? `
                            <div class="detail-section">
                                <div class="detail-title">Detailed Feedback</div>
                                <div class="detail-content">${file.detailed_feedback}</div>
                            </div>
                        ` : ''}
                        
                        ${file.recommendations && file.recommendations.length > 0 ? `
                            <div class="recommendations">
                                <div class="detail-title">Recommendations</div>
                                <ul>
                                    ${file.recommendations.map(rec => `<li>${rec}</li>`).join('')}
                                </ul>
                            </div>
                        ` : ''}
                    </div>
                `;
                
                container.appendChild(fileElement);
            });
        }

        function toggleFile(index) {
            const details = document.getElementById(`details-${index}`);
            const icon = document.getElementById(`icon-${index}`);
            
            details.classList.toggle('expanded');
            icon.classList.toggle('expanded');
        }

        function filterFiles(status) {
            currentFilter = status;
            
            // Update button states
            document.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active'));
            event.target.classList.add('active');
            
            // Apply filter
            if (status === 'all') {
                filteredFiles = [...fileData];
            } else {
                const statusMap = {
                    'ready': 'ready',
                    'conditional': 'conditional', 
                    'not_ready': 'not_ready'
                };
                filteredFiles = fileData.filter(file => file.merge_readiness === statusMap[status]);
            }
            
            renderFiles();
        }

        function searchFiles(query) {
            if (!query.trim()) {
                filterFiles(currentFilter);
                return;
            }
            
            const searchTerm = query.toLowerCase();
            filteredFiles = fileData.filter(file => 
                file.file_path.toLowerCase().includes(searchTerm) ||
                (file.overall_assessment && file.overall_assessment.purpose && 
                 file.overall_assessment.purpose.toLowerCase().includes(searchTerm))
            );
            
            renderFiles();
        }

        // Initialize on page load
        document.addEventListener('DOMContentLoaded', function() {
            if (typeof fileData !== 'undefined' && fileData.length > 0) {
                filteredFiles = [...fileData];
                renderFiles();
            }
        });
    </script>
