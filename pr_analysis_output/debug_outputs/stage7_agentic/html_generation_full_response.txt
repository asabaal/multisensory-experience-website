        function showTab(tabName) {
            // Hide all tab contents
            const contents = document.querySelectorAll('.tab-content');
            contents.forEach(content => content.classList.remove('active'));
            
            // Remove active class from all tabs
            const tabs = document.querySelectorAll('.tab');
            tabs.forEach(tab => tab.classList.remove('active'));
            
            // Show selected tab content
            document.getElementById(tabName).classList.add('active');
            
            // Add active class to clicked tab
            event.target.classList.add('active');
        }

        function toggleIssue(header) {
            const content = header.nextElementSibling;
            const card = header.parentElement;
            
            content.classList.toggle('expanded');
            card.classList.toggle('expanded');
        }

        function toggleDuplicate(header) {
            const content = header.nextElementSibling;
            const category = header.parentElement;
            
            content.classList.toggle('expanded');
            category.classList.toggle('expanded');
        }

        // Search functionality
        document.getElementById('searchInput').addEventListener('input', function(e) {
            const searchTerm = e.target.value.toLowerCase();
            
            // Search in issues
            const issueCards = document.querySelectorAll('.issue-card');
            issueCards.forEach(card => {
                const title = card.querySelector('.issue-title').textContent.toLowerCase();
                const content = card.querySelector('.issue-content').textContent.toLowerCase();
                
                if (title.includes(searchTerm) || content.includes(searchTerm)) {
                    card.classList.remove('hidden');
                } else {
                    card.classList.add('hidden');
                }
            });
            
            // Search in duplicates
            const duplicateCategories = document.querySelectorAll('.duplicate-category');
            duplicateCategories.forEach(category => {
                const title = category.querySelector('h3').textContent.toLowerCase();
                const content = category.querySelector('.duplicate-content').textContent.toLowerCase();
                
                if (title.includes(searchTerm) || content.includes(searchTerm)) {
                    category.classList.remove('hidden');
                } else {
                    category.classList.add('hidden');
                }
            });
            
            // Search in recommendations
            const recommendations = document.querySelectorAll('.recommendations-list li');
            recommendations.forEach(rec => {
                const text = rec.textContent.toLowerCase();
                
                if (text.includes(searchTerm)) {
                    rec.classList.remove('hidden');
                } else {
                    rec.classList.add('hidden');
                }
            });
        });

        // Initialize expand/collapse for duplicates
        document.addEventListener('DOMContentLoaded', function() {
            // Auto-expand critical duplicates
            const criticalDuplicates = document.querySelector('.severity-critical').closest('.duplicate-category');
            if (criticalDuplicates) {
                criticalDuplicates.classList.add('expanded');
                criticalDuplicates.querySelector('.duplicate-content').classList.add('expanded');
            }
        });
    </script>
